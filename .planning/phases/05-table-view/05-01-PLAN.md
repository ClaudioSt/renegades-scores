```javascript
function renderStandingsTable(entry, teamIds) {
  if (!entry.rows || entry.rows.length === 0) {
    return '<div class="standings-empty">Keine Tabellendaten verfügbar.</div>';
  }
  var thead = '<thead><tr>'
    + '<th class="col-team">Mannschaft</th>'
    + '<th>Sp</th><th>S</th><th>U</th><th>N</th>'
    + '<th>EP</th><th>GP</th><th>PD</th><th>SQ</th>'
    + '</tr></thead>';
  var restricted = entry.promotion_restricted || [];
  var rows = entry.rows.map(function(row) {
    var cls = '';
    if (teamIds.indexOf(row.team_id) >= 0) cls += ' standings-row-highlight';
    if (restricted.indexOf(row.team_id) >= 0) cls += ' standings-row-restricted';
    return '<tr class="' + cls.trim() + '">'
      + '<td class="col-team">' + escapeHtml(row.team_name) + '</td>'
      + '<td>' + row.Sp + '</td>'
      + '<td>' + row.S  + '</td>'
      + '<td>' + row.U  + '</td>'
      + '<td>' + row.N  + '</td>'
      + '<td>' + row.EP + '</td>'
      + '<td>' + row.GP + '</td>'
      + '<td>' + row.PD + '</td>'
      + '<td>' + row.SQ.toFixed(4) + '</td>'
      + '</tr>';
  }).join('');
  return '<table class="standings-table">' + thead + '<tbody>' + rows + '</tbody></table>';
}
```

**Change 2 — Replace the empty `renderTableView` stub** with an async version:

```javascript
async function renderTableView(root, cfg) {
  root.innerHTML = '<div class="loading">Lade Tabelle…</div>';
  var snap = await loadSnapshot();
  var standings = snap.standings || {};

  // Collect all {leagueKey, seasonKey, year, entry} tuples that have rows
  var allEntries = [];
  Object.keys(standings).forEach(function(lk) {
    Object.keys(standings[lk]).forEach(function(sk) {
      var entry = standings[lk][sk];
      if (entry.rows && entry.rows.length > 0) {
        allEntries.push({ leagueKey: lk, seasonKey: sk, year: parseInt(sk, 10) || 0, entry: entry });
      }
    });
  });

  // Find years where at least one cfg.team appears; fall back to all non-empty years
  var teamYears = allEntries
    .filter(function(e) {
      return e.entry.rows.some(function(r) { return cfg.teams.indexOf(r.team_id) >= 0; });
    })
    .map(function(e) { return e.year; });

  var activeYears = teamYears.length > 0
    ? teamYears.filter(function(y, i, a) { return a.indexOf(y) === i; })
    : allEntries.map(function(e) { return e.year; }).filter(function(y, i, a) { return a.indexOf(y) === i; });

  if (activeYears.length === 0) {
    root.innerHTML = '<div class="standings-empty">Keine Tabellendaten verfügbar.</div>';
    return;
  }

  // Default to most recent year
  activeYears.sort(function(a, b) { return b - a; });
  var defaultYear = activeYears[0];

  // Render tables for default year
  var entriesForYear = allEntries.filter(function(e) { return e.year === defaultYear; });
  var html = '<div class="standings-wrapper">';
  entriesForYear.forEach(function(e) {
    html += '<div class="standings-league-title">' + escapeHtml(e.entry.name) + '</div>';
    html += renderStandingsTable(e.entry, cfg.teams);
  });
  html += '</div>';

  root.innerHTML = html;
}
```
