    // ── VIEW RENDERERS ───────────────────────────────────────────────────────────
    async function renderSpielplan(root, cfg) {
      var elMap = {};
      cfg.teams.forEach(function(teamId) {
        var block = document.createElement('div');
        block.className = 'team-block';
        block.innerHTML =
          (cfg.showTitle ? '<div class="team-title" id="t-title-' + teamId + '"></div>' : '')
          + '<div id="t-live-' + teamId + '"></div>'
          + (cfg.showPast
              ? '<div class="section-title">Vergangene Spieltage</div>'
                + '<div id="t-past-' + teamId + '"><div class="loading">Lade Daten…</div></div>'
              : '')
          + (cfg.showFuture
              ? '<div class="section-title">Kommende Spieltage</div>'
                + '<div id="t-future-' + teamId + '"><div class="loading">Lade Daten…</div></div>'
              : '');
        root.appendChild(block);
        elMap[teamId] = {
          title:  document.getElementById('t-title-'  + teamId),
          live:   document.getElementById('t-live-'   + teamId),
          past:   document.getElementById('t-past-'   + teamId),
          future: document.getElementById('t-future-' + teamId),
        };
      });

      await Promise.all(cfg.teams.map(async function(teamId) {
        try {
          await loadTeam(teamId, elMap[teamId], cfg);
        } catch(err) {
          console.error('Widget error (team ' + teamId + '):', err);
          var errBanner = '<div class="error-banner">Fehler beim Laden: ' + escapeHtml(err.message) + '</div>';
          if (elMap[teamId].future) elMap[teamId].future.innerHTML = errBanner;
          if (elMap[teamId].past && elMap[teamId].past.querySelector('.loading'))
            elMap[teamId].past.innerHTML = errBanner;
        }
      }));
    }

    function renderTableView(root, cfg) {
      root.innerHTML = '';
    }
    ```

    // ── INIT ─────────────────────────────────────────────────────────────────────
    (async function() {

      document.documentElement.style.setProperty('--accent', '#' + cfg.color);
      if (cfg.compact) document.body.classList.add('compact');

      if (cfg.teams.length === 0) {

      if (cfg.view === 'table') {
        renderTableView(root, cfg);
      } else {
        await renderSpielplan(root, cfg);
      }
    })();
    ```
